#### 0. 示例json文件

```json
{
  "name":"Tom",
  "age":30,
  "company":{
    "name":"xxx"
  },
  "skills":[
    {
      "name":"Java",
      "level":3
    },
    {
      "name":"Go",
      "level":2
    },
    {
      "name":"Python",
      "level":3
    }
  ]
}
```
#### 1. 读

```
# 格式化打印
$ jq . json  
{
  "name": "Tom",
  "age": 30,
  "company": {
    "name": "xxx"
  },
  "skills": [
    {
      "name": "Java",
      "level": 3
    },
    {
      "name": "Go",
      "level": 2
    },
    {
      "name": "Python",
      "level": 3
    }
  ]
}

# 判断是否包含某个key
$ jq -r 'has("name")' json   
true

# 取name值
$ jq -r '.name' json
Tom

# 取company.name值
$ jq -r '.company.name' json
xxx
```

#### 2. 写

```
# 设置company.name值为ABC
$ jq  '.company.name="ABC"' | jq -r '.company.name' jso
ABC

# 修改company.name值，并保存到原json文件
$ jq  '.company.name="ABC"' json | jq . > json
```

#### 3. 数组

```
# 数组长度
$ jq -r '.skills | length ' json                                     
3

# 按键值过滤数组元素
$ jq -r '.skills | map(select(.name=="Java")) | .[0]' json
{
  "name": "Java",
  "level": 3
}

$ jq -r '.skills | map(select(.level>=3))' json       
[
  {
    "name": "Java",
    "level": 3
  },
  {
    "name": "Python",
    "level": 3
  }
]
```
